function computeWeights(drops, mf) {
  const tokenMain = drops.filter(d => d.pool !== 'EXTRA').reduce((s,d) => s+d.weight, 0);
  const allW = drops.reduce((s,d) => s+d.weight, 0);
  const baseChances = {};
  drops.forEach(d => {
    baseChances[d.name] = d.pool !== 'EXTRA' ? d.weight/tokenMain : d.weight/allW;
  });
  const mfW = {};
  drops.forEach(d => {
    mfW[d.name] = baseChances[d.name] < 0.05 ? d.weight*(1+mf/100) : d.weight*1.0;
  });
  const mfTM = drops.filter(d => d.pool !== 'EXTRA').reduce((s,d) => s+mfW[d.name], 0);
  const mfAll = drops.reduce((s,d) => s+mfW[d.name], 0);
  return { mfW, mfTM, mfAll, baseChances };
}

function solveOptimal(drop, drops, mf, pityLevel, aatroxOn) {
  const { mfW, mfTM, mfAll, baseChances } = computeWeights(drops, mf);
  const baseChancePct = baseChances[drop.name] * 100;
  const reqXp = Math.floor(50000 / baseChancePct);
  const aatroxMult = aatroxOn ? 1.25 : 1.0;
  const pityMult = 1 + pityLevel * 0.01;
  const xpPerRun = currentBoss.base_xp * aatroxMult * pityMult;
  const m = Math.ceil(reqXp / xpPerRun);
  const baseTW = mfW[drop.name];
  const poolTotal = drop.pool !== 'EXTRA' ? mfTM : mfAll;
  const p = baseTW / poolTotal;

  let bestL = 0, maxEff = 0, currentS = 1.0, sumS = 0.0;
  const effByL = [];

  for (let L = 0; L <= m; L++) {
    const num = 1 + currentS * (m - L) * p;
    const den = sumS + currentS * (m - L + 1);
    const eff = num / den;
    effByL.push({ L, eff, pct: (L / m) * 100 });
    if (eff > maxEff) { maxEff = eff; bestL = L; }

    if (L < m) {
      const storedXp = L * xpPerRun;
      const mult = 1 + Math.min(2 * (storedXp / reqXp), 2);
      const boostedW = baseTW * mult;
      const curTotal = drop.pool !== 'EXTRA' ? mfTM - baseTW + boostedW : mfAll - baseTW + boostedW;
      const pL = boostedW / curTotal;
      sumS += currentS;
      currentS *= (1 - pL);
    }
  }

  const baselineEff = (1 + m * p) / (m + 1);
  const gain = ((maxEff / baselineEff) - 1) * 100;
  const baseChanceNoMF = baseChances[drop.name];

  return { bestL, maxEff, m, p, baseChanceNoMF, reqXp, xpPerRun, baselineEff, gain, effByL, pct: (bestL/m)*100 };
}

function buildDropGrid() {
  const grid = document.getElementById('drop-grid');
  grid.innerHTML = '';
  const { baseChances } = computeWeights(currentBoss.drops, 0);
  currentBoss.drops.forEach((d, i) => {
    const pct = baseChances[d.name] * 100;
    const btn = document.createElement('button');
    btn.className = 'drop-btn' + (selectedDrop === i ? ' active' : '');
    const poolClass = d.pool === 'MAIN' ? 'pool-main' : d.pool === 'EXTRA' ? 'pool-extra' : 'pool-token';
    btn.innerHTML = `<div class="drop-name">${d.name}</div> 
      <div class="drop-chance">~${pct < 0.1 ? pct.toFixed(3) : pct.toFixed(2)}% base</div> 
      <div class="drop-pool ${poolClass}">${d.pool}</div>`;
    btn.onclick = () => { selectedDrop = i; buildDropGrid(); compute(); };
    grid.appendChild(btn);
  });
}

function compute() {
  if (selectedDrop === null) return;
  const drop = currentBoss.drops[selectedDrop];
  const mf = parseInt(document.getElementById('sl-mf').value);
  const pity = parseInt(document.getElementById('sl-pity').value);
  const res = solveOptimal(drop, currentBoss.drops, mf, pity, aatrox);

  document.getElementById('results-section').style.display = '';
  document.getElementById('no-target').style.display = 'none';

  document.getElementById('r-req-xp').textContent = res.reqXp.toLocaleString();
  document.getElementById('r-m').textContent = res.m.toLocaleString();
  document.getElementById('r-xp-run').textContent = res.xpPerRun.toFixed(0) + ' XP/run';
  document.getElementById('r-base-chance').textContent = (res.baseChanceNoMF*100).toFixed(4) + '%';
  document.getElementById('r-base-1in').textContent = '1 in ' + Math.round(1/res.baseChanceNoMF).toLocaleString();
  document.getElementById('r-mf-chance').textContent = (res.p*100).toFixed(4) + '%';
  document.getElementById('r-mf-1in').textContent = '1 in ' + Math.round(1/res.p).toLocaleString();

  document.getElementById('r-best-l').textContent = res.bestL.toLocaleString() + ' runs';
  document.getElementById('r-best-pct').textContent = res.pct.toFixed(1) + '%';
  document.getElementById('r-gain').textContent = '+' + res.gain.toFixed(2) + '%';
  document.getElementById('r-opt-rate').textContent = '1 in ' + Math.round(1/res.maxEff).toLocaleString();
  document.getElementById('r-base-rate').textContent = '1 in ' + Math.round(1/res.baselineEff).toLocaleString();

  const pBar = document.getElementById('progress-bar');
  pBar.style.width = Math.min(res.pct, 100).toFixed(1) + '%';
  document.getElementById('prog-label').textContent = res.pct.toFixed(1) + '%';

  buildAdvice(drop, res, mf);
  buildStrategy(drop, res, mf, pity);

  if (currentTab === 'curve') renderChart(res);
  window._lastRes = res;
}

function buildAdvice(drop, res, mf) {
  const box = document.getElementById('advice-box');
  const txt = document.getElementById('advice-text');
  let msg = '';
  const pct = res.pct;

  if (res.bestL === 0) {
    msg = `For <strong>${drop.name}</strong>, the optimal strategy is to <strong>never turn the meter ON</strong> (L*=0). This happens because XP per run is high and the marginal gain from the meter does not offset the reset risk. Let it accumulate and use the pity guarantee.`;
    box.className = 'warn-card';
  } else if (pct <= 40) {
    msg = `Keep the meter ON only for the <strong>first ${res.bestL} runs (${pct.toFixed(0)}% of the meter)</strong>, then turn it off. Early meter boosts drop chance without risking much accumulated progress.`;
    box.className = 'advice-box';
  } else if (pct <= 70) {
    msg = `The sweet spot is at <strong>${pct.toFixed(0)}% of the meter (${res.bestL} runs)</strong>. Keep the meter ON normally and turn it off when you reach this point. This gives <strong>+${res.gain.toFixed(1)}%</strong> more drops compared to leaving it always off.`;
    box.className = 'advice-box';
  } else {
    msg = `With your Magic Find of <strong>${mf}</strong>, the meter is worth keeping ON until <strong>${pct.toFixed(0)}%</strong>. High MF shifts the sweet spot later because the per-run chance boost compensates more for resets.`;
    box.className = 'advice-box';
  }
  txt.innerHTML = msg;
}

function renderChart(res) {
  res = res || window._lastRes;
  if (!res) return;
  const labels = res.effByL.map(x => x.L);
  const vals = res.effByL.map(x => (1/x.eff).toFixed(1));
  const colors = res.effByL.map(x => x.L === res.bestL ? '#E24B4A' : '#1D9E75');

  if (effChart) { effChart.destroy(); effChart = null; }

  const ctx = document.getElementById('eff-chart').getContext('2d');
  effChart = new Chart(ctx, {
    type: 'line',
    data: {
      labels,
      datasets: [{
        label: '1 in N runs for drop',
        data: res.effByL.map(x => parseFloat((1/x.eff).toFixed(1))),
        borderColor: '#1D9E75',
        backgroundColor: 'rgba(29,158,117,0.08)',
        fill: true,
        pointRadius: res.effByL.map(x => x.L === res.bestL ? 6 : 2),
        pointBackgroundColor: colors,
        pointBorderColor: colors,
        tension: 0.3,
        borderWidth: 2
      }]
    },
    options: {
      responsive: true, maintainAspectRatio: false,
      plugins: {
        legend: { display: false },
        tooltip: {
          callbacks: {
            label: ctx => `1 in ${ctx.parsed.y} runs | ${((ctx.dataIndex/res.m)*100).toFixed(0)}% of meter`
          }
        }
      },
      scales: {
        x: {
          title: { display: true, text: 'Runs with meter ON (L)', color: '#888780', font: { size: 11 } },
          ticks: { maxTicksLimit: 10, color: '#888780' },
          grid: { color: 'rgba(136,135,128,0.15)' }
        },
        y: {
          title: { display: true, text: '1 in N runs (lower = better)', color: '#888780', font: { size: 11 } },
          ticks: { color: '#888780' },
          grid: { color: 'rgba(136,135,128,0.15)' }
        }
      }
    }
  });
}
